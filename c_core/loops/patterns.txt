#include <stdio.h>

int main() {
    
    int num = 5;
    
    for (int row = 1; row<=num; row++){
        for (int col = 1; col<=num; col++){
            printf("* ");
        }
        printf("\n");
    }
    return 0;

#include <stdio.h>

int main() {
    
    int num = 5;
    
    for (int row = 1; row<=num; row++){
        for (int col = num; col>row; col--){
            printf("  ");
        }
        for (int col = num; col>=num-row+1; col--){
            printf("* ");
        }
        printf("\n");
    }
    
    return 0;
}

        for (int col = 1; col<=num; col++){
            if(row == col){
                printf("* ");

                printf("- ");

        }
