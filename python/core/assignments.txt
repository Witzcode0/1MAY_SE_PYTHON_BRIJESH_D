# 0 1 1 2 3 5 8 13

num = 10
n1 = 0
n2 = 1

for no in range(num):
    if no == 0:
        print(0, end=" ")
    elif no == 1:
        print(1, end=" ")
    else:
        nth = n1 + n2
        print(nth, end=" ")
        n1 = n2
        n2 = nth
